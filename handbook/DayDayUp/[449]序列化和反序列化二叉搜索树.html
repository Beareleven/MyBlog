    449. 序列化和反序列化二叉搜索树

    Bear 2022/5/11

    序列化是将数据结构或对象转换为一系列位的过程，以便它可以存储在文件或内存缓冲区中，或通过网络连接链路传输，以便稍后在同一个或另一个计算机环境中重建。 设计一个算法来序列化和反序列化 二叉搜索树 。 对序列化/反序列化算法的工作方式没有限制。 您只需确保二叉搜索树可以序列化为字符串，并且可以将该字符串反序列化为最初的二叉搜索树。 编码的字符串应尽可能紧凑。

    # 构建二叉树

    # 105. 从中序与前序遍历序列构造二叉树

    /**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} preorder
 * @param {number[]} inorder
 * @return {TreeNode}
 */
 // 105.  从中序与前序遍历序列构造二叉树
var buildTree = function(preorder, inorder) {
    // 判空
    if(!preorder.length) return null;

    // 第一个节点即是根结点
    let rootVal = preorder.shift();
    let root = new TreeNode(rootVal);

    // 获取切割下标
    let index = 0;
    for(index; index < inorder.length; index++){
        if(inorder[index] === rootVal) break;
    }

    // 创建左右子树
    root.left = buildTree(preorder.slice(0, index), inorder.slice(0,index));
    root.right = buildTree(preorder.slice(index), inorder.slice(index + 1));

    return root;
};

    # 106. 从中序与后序遍历序列构造二叉树

    /**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} inorder
 * @param {number[]} postorder
 * @return {TreeNode}
 */
 // 106. 从中序与后序遍历序列构造二叉树
var buildTree = function(inorder, postorder) {
    // 后序遍历的最后一个值是根结点，中序遍历的根节点左边是左子树，右边是右子树
    
    // 1. 判空
    if(!postorder.length) return null;

    // 2. 后序遍历的最后一个值是根结点
    let rootVal = postorder.pop();
    let root = new TreeNode(rootVal);

    // 4. 切割左子树出来,
    let inorderLen = inorder.length
    // 获取中间节点在中序遍历下的下标
    let index = 0
    for(index; index < inorderLen; index++){
        if(inorder[index] === rootVal) break;
    }

    // 创建左右子树
    // 在中序遍历中，中间节点的左边是他的左子树，右边是右子树
    // 在后序遍历中，切割出与中序遍历同样多节点的节点数，对应是就是他左子树的后序遍历
    root.left = buildTree(inorder.slice(0, index), postorder.slice(0, index));
    root.right = buildTree(inorder.slice(index + 1), postorder.slice(index));

    return root;
};

    # 449. 序列化和反序列化二叉搜索树

    /**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
// 449. 序列化和反序列化二叉搜索树
/**
 * Encodes a tree to a single string.
 *
 * @param {TreeNode} root
 * @return {string}
 */
var serialize = function(root) {

    let preorder = [], inorder = [];
    // 前序遍历
    function traverTree(root){
        if(!root) return;
        preorder.push(root.val);
        traverTree(root.left);
        traverTree(root.right);
    }

    traverTree(root);
    inorder = [...preorder].sort((a,b) => a - b);
    // console.log(preorder, inorder)
    return [preorder, inorder];
};
/**
 * Decodes your encoded data to tree.
 *
 * @param {string} data
 * @return {TreeNode}
 */
var deserialize = function(data) {
    // console.log(data)
    // data[0] = preorder, data[1] = inorder;
    let preorder = data[0], inorder = data[1];
    // console.log(preorder, inorder)
    if(!preorder.length) return null;

    let rootVal = preorder.shift();
    let root = new TreeNode(rootVal);

    let index = 0;
    for(index; index < inorder.length; index++){
        if(inorder[index] === rootVal) break;
    }

    root.left = deserialize([preorder.slice(0, index), inorder.slice(0, index)]);
    root.right = deserialize([preorder.slice(index), inorder.slice(index + 1)]);

    return root;
};
